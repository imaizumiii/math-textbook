\section*{Complex TikZ Example}


\noindent 関数グラフの描画例：


\begin{figure}[h]
    \centering
    \begin{tikzpicture}

        \draw[->] (-0.5,0) -- (4.5,0) node[right] {$x$};
        \draw[->] (0,-0.5) -- (0,4.5) node[above] {$y$};
        \draw[scale=0.5,domain=0:4,smooth,variable=\x,blue] plot ({\x},{\x*\x});
        
\end{tikzpicture}
    \caption{Function Graph}
    \label{fig:graph}
\end{figure}
